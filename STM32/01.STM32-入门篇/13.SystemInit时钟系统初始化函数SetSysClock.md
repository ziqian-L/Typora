（2）删除后的代码（使用的正点原子的代码，和视频不一样，删除后是一样的）

```c
static void SetSysClock(void)
{
/******************************************************************************/
/*            PLL (clocked by HSE) used as System clock source                */
/******************************************************************************/
  __IO uint32_t StartUpCounter = 0, HSEStatus = 0;
  
  /* Enable HSE */
  /*使能HSE*/
  RCC->CR |= ((uint32_t)RCC_CR_HSEON);
 
  /* Wait till HSE is ready and if Time out is reached exit */
  /*等待HSE启动稳定，如果超时则退出*/
  do
  {
    HSEStatus = RCC->CR & RCC_CR_HSERDY;
    StartUpCounter++;
  } while((HSEStatus == 0) && (StartUpCounter != HSE_STARTUP_TIMEOUT));

  if ((RCC->CR & RCC_CR_HSERDY) != RESET)
  {
    HSEStatus = (uint32_t)0x01;
  }
  else
  {
    HSEStatus = (uint32_t)0x00;
  }

/*HSE启动成功*/
  if (HSEStatus == (uint32_t)0x01)
  {
    /* Select regulator voltage output Scale 1 mode */
    /*选择电压调节器的模式为1*/
    RCC->APB1ENR |= RCC_APB1ENR_PWREN;
    PWR->CR |= PWR_CR_VOS;

    /* HCLK = SYSCLK / 1*/
    RCC->CFGR |= RCC_CFGR_HPRE_DIV1;//控制 AHB 时钟分频系数为1
    /* PCLK2 = HCLK / 2*/
    RCC->CFGR |= RCC_CFGR_PPRE2_DIV2;//控制 APB2 时钟分频系数为2
    /* PCLK1 = HCLK / 4*/
    RCC->CFGR |= RCC_CFGR_PPRE1_DIV4;//控制 APB1 时钟分频系数为4
   
    /* Configure the main PLL */
    /*配置主PLL，选择锁相环时钟来源*/
    /*配置分频系数M | 配置倍频系数N | 配置分频系数P | 配置分频系数Q*/
    /*407没有分频系数R，所以不用配置，它在441里有*/
    RCC->PLLCFGR = PLL_M | (PLL_N << 6) | (((PLL_P >> 1) -1) << 16) |
                   (RCC_PLLCFGR_PLLSRC_HSE) | (PLL_Q << 24);

    /* Enable the main PLL */
    /*使能主PLL*/
    RCC->CR |= RCC_CR_PLLON;

    /* Wait till the main PLL is ready */
    /*等待主PLL稳定*/
    while((RCC->CR & RCC_CR_PLLRDY) == 0)
    {
    }
  
    /* Configure Flash prefetch, Instruction cache, Data cache and wait state */
    /*配置FLASH预取指。，指令缓存，数据缓存，等待周期*/
    FLASH->ACR = FLASH_ACR_PRFTEN | FLASH_ACR_ICEN |FLASH_ACR_DCEN |FLASH_ACR_LATENCY_5WS;

    /* Select the main PLL as system clock source */
    /*选择主PLL时钟作为系统时钟*/
    RCC->CFGR &= (uint32_t)((uint32_t)~(RCC_CFGR_SW));
    RCC->CFGR |= RCC_CFGR_SW_PLL;

    /* Wait till the main PLL is used as system clock source */
    /*确保主PLL时钟选为系统时钟*/
    while ((RCC->CFGR & (uint32_t)RCC_CFGR_SWS ) != RCC_CFGR_SWS_PLL);
    {
    }
  }
  else
  { /* If HSE fails to start-up, the application will have wrong clock
         configuration. User can add here some code to deal with this error */
    /*如果HSE启动失败，在这里添加启动失败的处理代码*/
  }
}
```

（3）代码工作原理

1. 代码作用：配置系统时间源，PLL倍频器和分频器因子，AHB/APBX预分频器和FLASH相关设置 

2. 流程：

      ​														经典值
   HSE（高速外部时钟）经过分频器==除以M倍==进入主PLL（锁相环） 8MHz / 8 = 1MHz
      					$\downarrow$
	过倍频器==乘N倍==变成VCO时钟								1MHz * 336 = 336MHz
      					$\downarrow$
	CO时钟经过分频器==除以P倍==输出锁相环时钟					336MHz /2 = 168MHz
      					$\downarrow$
   锁相环时钟被选择为系统时钟
      					$\downarrow$
	系统时钟经过==分频==后成为c							168MHZ / 1 = 168MHz
	系统时钟经过==分频==后成为APB1总线时钟						168MHz / 4 = 42MHz
	系统时钟经过==分频==后成为APB2总线时钟						168MHz / 2 = 84MHz

   这三根总线的时钟为最大值为AHB=168MHz，APB1=42MHz，APB2=84MHz
